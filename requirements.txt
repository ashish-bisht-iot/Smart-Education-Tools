# -*- coding: utf-8 -*-
import sys, io
# Force UTF-8 output on Windows so arrow/special chars don't crash
sys.stdout = io.TextIOWrapper(sys.stdout.buffer, encoding='utf-8', errors='replace')

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.ticker as mticker
import matplotlib.colors as mcolors
import matplotlib.cm as cm

# ── 1. Load & prep ─────────────────────────────────────────────────────────────
df = pd.read_csv('cleaned_amazon_dataset.csv')
df['OrderDate'] = pd.to_datetime(df['OrderDate'], dayfirst=True)
df['Country'] = df['Country'].str.strip().str.title()
df['State']   = df['State'].str.strip().str.title()
if 'City' in df.columns:
    df['City'] = df['City'].str.strip().str.title()

# ── 2. Country summary ─────────────────────────────────────────────────────────
country_summary = df.groupby('Country').agg(
    Total_Sales     = ('TotalAmount',    'sum'),
    Avg_Order_Value = ('TotalAmount',    'mean'),
    Total_Orders    = ('OrderID',        'count'),
    Avg_Shipping    = ('ShippingCost',   'mean'),
    Logistics_Ratio = ('LogisticsRatio', 'mean'),
    Cancel_Rate     = ('IsDelivered',    lambda x: (1 - x.mean()) * 100)
).round(2).sort_values('Total_Sales', ascending=False)
print("=== Country Summary ===")
print(country_summary)


# -- State name -> USPS abbreviation (defined early so we can expand before groupby)
STATE_ABBR = {
    'Alabama':'AL','Alaska':'AK','Arizona':'AZ','Arkansas':'AR','California':'CA',
    'Colorado':'CO','Connecticut':'CT','Delaware':'DE','Florida':'FL','Georgia':'GA',
    'Hawaii':'HI','Idaho':'ID','Illinois':'IL','Indiana':'IN','Iowa':'IA',
    'Kansas':'KS','Kentucky':'KY','Louisiana':'LA','Maine':'ME','Maryland':'MD',
    'Massachusetts':'MA','Michigan':'MI','Minnesota':'MN','Mississippi':'MS',
    'Missouri':'MO','Montana':'MT','Nebraska':'NE','Nevada':'NV','New Hampshire':'NH',
    'New Jersey':'NJ','New Mexico':'NM','New York':'NY','North Carolina':'NC',
    'North Dakota':'ND','Ohio':'OH','Oklahoma':'OK','Oregon':'OR','Pennsylvania':'PA',
    'Rhode Island':'RI','South Carolina':'SC','South Dakota':'SD','Tennessee':'TN',
    'Texas':'TX','Utah':'UT','Vermont':'VT','Virginia':'VA','Washington':'WA',
    'West Virginia':'WV','Wisconsin':'WI','Wyoming':'WY','District Of Columbia':'DC',
}

# Reverse map: abbreviation -> full name (fixes CSVs that store e.g. "Ca", "Tx")
ABBR_STATE = {v: k for k, v in STATE_ABBR.items()}

def expand_state(s):
    upper = str(s).strip().upper()
    return ABBR_STATE.get(upper, s)  # convert abbr -> full name if matched

# FIX: expand BEFORE filtering/grouping so state names match GeoJSON
df['State'] = df['State'].apply(expand_state)

# -- 3. US filter ---
us_df = df[df['Country'] == 'United States'].copy()

# -- 4. State summary ---
state_summary = us_df.groupby('State').agg(
    Total_Sales     = ('TotalAmount',    'sum'),
    Avg_Order_Value = ('TotalAmount',    'mean'),
    Total_Orders    = ('OrderID',        'count'),
    Avg_Shipping    = ('ShippingCost',   'mean'),
    Logistics_Ratio = ('LogisticsRatio', 'mean'),
    Cancel_Rate     = ('IsDelivered',    lambda x: (1 - x.mean()) * 100)
).round(2).sort_values('Total_Sales', ascending=False)
print("\n=== Top 10 States by Sales ===")
print(state_summary.head(10))

# ── 5. Choropleth ──────────────────────────────────────────────────────────────
try:
    import geopandas as gpd

    url = ("https://raw.githubusercontent.com/PublicaMundi/MappingAPI/"
           "master/data/geojson/us-states.json")
    states_geo = gpd.read_file(url)

    states_geo['name_key'] = states_geo['name'].str.strip().str.title()

    ss_merge = state_summary[['Total_Sales','Total_Orders','Cancel_Rate']].reset_index()
    ss_merge.columns = ['name_key','Total_Sales','Total_Orders','Cancel_Rate']

    geo_names  = set(states_geo['name_key'])
    data_names = set(ss_merge['name_key'])
    unmatched  = data_names - geo_names
    if unmatched:
        print(f"[WARN] States in data not matched to GeoJSON: {sorted(unmatched)}")

    merged = states_geo.merge(ss_merge, on='name_key', how='left')
    matched = merged['Total_Sales'].notna().sum()
    print(f"[INFO] {matched}/{len(states_geo)} states have data on map")

    # ── Impute missing states with national averages so the map is fully coloured
    # Mark which rows are real data vs imputed (used for hatching later)
    merged['_has_data'] = merged['Total_Sales'].notna()

    nat_avg_sales   = ss_merge['Total_Sales'].mean()
    nat_avg_orders  = ss_merge['Total_Orders'].mean()
    nat_avg_cancel  = ss_merge['Cancel_Rate'].mean()

    merged['Total_Sales']  = merged['Total_Sales'].fillna(nat_avg_sales)
    merged['Total_Orders'] = merged['Total_Orders'].fillna(nat_avg_orders)
    merged['Cancel_Rate']  = merged['Cancel_Rate'].fillna(nat_avg_cancel)

    missing_states = merged.loc[~merged['_has_data'], 'name_key'].tolist()
    print(f"[INFO] {len(missing_states)} states filled with national average (shown hatched): "
          f"{sorted(missing_states)}")

    metrics = [
        ('Total_Sales',  'Total Sales ($)',        'YlOrRd'),
        ('Total_Orders', 'Total Orders',           'Blues'),
        ('Cancel_Rate',  'Cancellation Rate (%)', 'RdPu'),
    ]
    fig, axes = plt.subplots(1, 3, figsize=(26, 8))
    fig.patch.set_facecolor('#F7F9FC')

    for ax, (col, title, cmap) in zip(axes, metrics):
        vmin = merged[col].quantile(0.05)
        vmax = merged[col].quantile(0.95)

        # Plot all states with colour
        merged.plot(
            column=col, ax=ax,
            cmap=cmap,
            vmin=vmin, vmax=vmax,
            legend=True,
            legend_kwds={
                'orientation': 'horizontal',
                'shrink': 0.7,
                'pad': 0.02,
                'label': title,
                'format': '%.0f'
            },
            edgecolor='white', linewidth=0.4,
        )

        # Overlay hatch on imputed (no-real-data) states so viewers know
        imputed_geom = merged[~merged['_has_data']]
        if not imputed_geom.empty:
            imputed_geom.plot(
                ax=ax,
                facecolor='none',
                edgecolor='#888',
                linewidth=0.5,
                hatch='////',
            )

        ax.set_title(title, fontsize=13, fontweight='bold', pad=8)
        ax.axis('off')

        # Label every state (real data = bold black, imputed = grey italic)
        for _, row in merged.iterrows():
            cx = row.geometry.centroid.x
            cy = row.geometry.centroid.y
            abbr = STATE_ABBR.get(row['name_key'], '')
            if not abbr:
                continue
            is_real = row['_has_data']
            ax.text(cx, cy, abbr,
                    fontsize=5, ha='center', va='center',
                    color='#111' if is_real else '#555',
                    fontweight='bold' if is_real else 'normal',
                    style='normal' if is_real else 'italic',
                    bbox=dict(boxstyle='round,pad=0.1', fc='white',
                              alpha=0.45 if is_real else 0.25, ec='none'))

    # Legend patch explaining hatch
    from matplotlib.patches import Patch
    hatch_patch = Patch(facecolor='white', edgecolor='#888', hatch='////',
                        label='National avg (no data in CSV)')
    axes[-1].legend(handles=[hatch_patch], loc='lower right',
                    fontsize=7, framealpha=0.7)

    plt.suptitle('Amazon US Sales — State-level Choropleth\n'
                 '(hatched = national average estimate, no raw data for that state)',
                 fontsize=14, fontweight='bold', y=1.01, color='#1a1a2e')
    plt.tight_layout()
    plt.savefig('amazon_us_choropleth.png', dpi=150, bbox_inches='tight',
                facecolor=fig.get_facecolor())
    plt.show()
    print("Choropleth saved -> amazon_us_choropleth.png")

# ── 5b. Fallback 3-panel dashboard (no geopandas needed) ──────────────────────
except ImportError:
    print("[INFO] geopandas not found – rendering 3-panel state dashboard")

    n      = len(state_summary)
    ss     = state_summary
    ss_asc = ss.sort_values('Total_Sales', ascending=True)
    asc_abbr = [STATE_ABBR.get(s, s[:2].upper()) for s in ss_asc.index]

    # ── FIX: Use GridSpec instead of add_axes to avoid colorbar layout conflicts
    fig = plt.figure(figsize=(24, max(14, n * 0.32) + 4))
    fig.patch.set_facecolor('#F7F9FC')

    from matplotlib.gridspec import GridSpec
    gs = GridSpec(2, 2, figure=fig,
                  left=0.05, right=0.97,
                  top=0.94, bottom=0.05,
                  hspace=0.35, wspace=0.30,
                  width_ratios=[1, 1.1])

    ax1 = fig.add_subplot(gs[:, 0])   # Full left column
    ax2 = fig.add_subplot(gs[0, 1])   # Top right
    ax3 = fig.add_subplot(gs[1, 1])   # Bottom right

    # ── Panel 1: Horizontal bar — Total Sales ────────────────────────────────
    sales_vals = ss_asc['Total_Sales'].values
    s_min, s_max = sales_vals.min(), sales_vals.max()

    # FIX: Normalize colors directly from actual sales values
    norm1     = mcolors.Normalize(vmin=s_min, vmax=s_max)
    cmap1     = cm.YlOrRd
    bar_colors = [cmap1(norm1(v)) for v in sales_vals]

    bars = ax1.barh(asc_abbr, sales_vals,
                    color=bar_colors, edgecolor='white', linewidth=0.3, height=0.78)

    ax1.xaxis.set_major_formatter(mticker.FuncFormatter(lambda x, _: f'${x/1e3:.0f}K'))
    ax1.set_xlabel('Total Sales', fontsize=9, labelpad=6)
    ax1.set_title('Total Sales – All US States', fontsize=12, fontweight='bold', pad=8)
    ax1.tick_params(axis='y', labelsize=7)
    ax1.tick_params(axis='x', labelsize=8)
    ax1.set_facecolor('#EAEFF5')
    ax1.spines[['top', 'right']].set_visible(False)
    ax1.grid(axis='x', linestyle='--', alpha=0.5, color='white')

    # Value labels on bars
    for bar, val in zip(bars, sales_vals):
        ax1.text(val + s_max * 0.012,
                 bar.get_y() + bar.get_height() / 2,
                 f'${val/1e3:.0f}K',
                 va='center', fontsize=6, color='#333')

    # FIX: Colorbar now correctly tied to the same norm/cmap used for bars
    sm1 = cm.ScalarMappable(cmap=cmap1, norm=norm1)
    sm1.set_array(sales_vals)
    cb1 = fig.colorbar(sm1, ax=ax1, shrink=0.30, pad=0.01,
                       orientation='horizontal', location='bottom')
    cb1.ax.tick_params(labelsize=7)
    cb1.set_label('Sales ($)', fontsize=8)
    cb1.ax.xaxis.set_major_formatter(
        mticker.FuncFormatter(lambda x, _: f'${x/1e3:.0f}K'))

    # ── Panel 2: Scatter — Sales vs Cancellation Rate ────────────────────────
    orders_vals = ss['Total_Orders'].values
    o_min, o_max = orders_vals.min(), orders_vals.max()

    # FIX: Use vmin/vmax with small padding to avoid flat colors when range is narrow
    o_range = o_max - o_min if o_max != o_min else 1
    norm2 = mcolors.Normalize(vmin=o_min - o_range * 0.1,
                               vmax=o_max + o_range * 0.1)

    sc = ax2.scatter(
        ss['Total_Sales'], ss['Cancel_Rate'],
        s=60 + (orders_vals / o_max) * 320,
        c=orders_vals,
        cmap='Blues',
        norm=norm2,
        alpha=0.88,
        edgecolors='#444',
        linewidth=0.5,
        zorder=3
    )

    for name, row in ss.iterrows():
        abbr = STATE_ABBR.get(name, name[:2].upper())
        ax2.annotate(abbr,
                     (row['Total_Sales'], row['Cancel_Rate']),
                     fontsize=6.5, ha='center', va='bottom',
                     xytext=(0, 4), textcoords='offset points',
                     color='#222')

    cb2 = fig.colorbar(sc, ax=ax2, shrink=0.80, pad=0.02)
    cb2.set_label('Total Orders', fontsize=8)
    cb2.ax.tick_params(labelsize=7)

    ax2.xaxis.set_major_formatter(
        mticker.FuncFormatter(lambda x, _: f'${x/1e3:.0f}K'))
    ax2.set_xlabel('Total Sales', fontsize=9)
    ax2.set_ylabel('Cancellation Rate (%)', fontsize=9)
    ax2.set_title('Sales vs Cancellation Rate\n(bubble size = order volume)',
                  fontsize=11, fontweight='bold', pad=6)
    ax2.set_facecolor('#EAEFF5')
    ax2.spines[['top', 'right']].set_visible(False)
    ax2.grid(linestyle='--', alpha=0.4)

    # ── Panel 3: Grouped bar — Avg Order Value vs Avg Shipping ───────────────
    top15   = ss.head(15).sort_values('Avg_Order_Value', ascending=True)
    t_abbrs = [STATE_ABBR.get(s, s[:2].upper()) for s in top15.index]
    y, h    = np.arange(len(top15)), 0.35

    ax3.barh(y + h / 2, top15['Avg_Order_Value'], h,
             color='#4C9BE8', alpha=0.90, label='Avg Order Value ($)')
    ax3.barh(y - h / 2, top15['Avg_Shipping'], h,
             color='#F4845F', alpha=0.90, label='Avg Shipping Cost ($)')

    ax3.set_yticks(y)
    ax3.set_yticklabels(t_abbrs, fontsize=8)
    ax3.xaxis.set_major_formatter(
        mticker.FuncFormatter(lambda x, _: f'${x:,.0f}'))
    ax3.set_xlabel('Amount ($)', fontsize=9)
    ax3.set_title('Avg Order Value vs Avg Shipping – Top 15 States',
                  fontsize=11, fontweight='bold', pad=6)
    ax3.legend(fontsize=8, loc='lower right')
    ax3.set_facecolor('#EAEFF5')
    ax3.spines[['top', 'right']].set_visible(False)
    ax3.grid(axis='x', linestyle='--', alpha=0.4)

    fig.suptitle('Amazon US Sales — State-level Dashboard',
                 fontsize=16, fontweight='bold', color='#1a1a2e')

    plt.savefig('amazon_us_state_dashboard.png', dpi=150, bbox_inches='tight',
                facecolor=fig.get_facecolor())
    plt.show()
    print("Dashboard saved -> amazon_us_state_dashboard.png")

# ── 6. City summary ────────────────────────────────────────────────────────────
city_summary = (
    us_df.groupby('City').agg(
        Total_Sales  = ('TotalAmount',  'sum'),
        Avg_Shipping = ('ShippingCost', 'mean'),
        Order_Count  = ('OrderID',      'count')
    ).round(2).sort_values('Total_Sales', ascending=False).head(15)
)
print("\n=== Top 15 Cities by Sales ===")
print(city_summary)

fig, ax = plt.subplots(figsize=(10, 7))
city_colors = plt.cm.Blues(np.linspace(0.4, 0.9, 15))[::-1]
city_summary['Total_Sales'].sort_values().plot(
    kind='barh', color=city_colors, ax=ax, edgecolor='white')
ax.xaxis.set_major_formatter(mticker.FuncFormatter(lambda x, _: f'${x:,.0f}'))
ax.set_xlabel('Total Sales ($)', fontsize=11)
ax.set_title('Top 15 US Cities by Sales', fontsize=14, fontweight='bold', pad=12)
ax.grid(axis='x', linestyle='--', alpha=0.4)
ax.spines[['top', 'right']].set_visible(False)
plt.tight_layout()
plt.savefig('amazon_top15_cities.png', dpi=150, bbox_inches='tight')
plt.show()
print("City chart saved -> amazon_top15_cities.png")
